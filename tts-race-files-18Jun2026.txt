   228	/**
   229	 * Attempt to reclaim a recovering session for a reconnecting device.
   230	 * Returns the reclaimed session, or null if no recovering session found.
   231	 *
   232	 * ATOMIC: clears recovery timer, increments connectionGeneration, sets
   233	 * state to "active", and attaches the new WS — all before returning.
   234	 * Logs CALL_RECONNECTED.

   236	export function reclaimSession(
   237	  deviceId: string,
   238	  ws: WsLike,
   239	): VoiceSession | null {
   240	  const session = findByDeviceId(deviceId);
   241	  if (!session || session.state !== "recovering") return null;
   242	
   243	  // Atomically clear timer + increment generation + attach new ws.
   244	  if (session.recoveryTimer !== null) {
   245	    clearTimeout(session.recoveryTimer);
   246	    session.recoveryTimer = null;
   247	  }

   252	  // P1-3: push timestamp into ring buffer, cap at 20 entries.

   282	 * Starts a 60-second timer that calls finalise() if no reconnect arrives.
   283	 * The timer re-checks state AND connectionGeneration before finalising
   284	 * so a concurrent reclaimSession() wins the race cleanly.
   285	 * Logs CALL_RECOVERING.

   287	export function markRecovering(sessionId: string): void {
   288	  const session = sessionsBySessionId.get(sessionId);
   289	  if (!session) return;
   290	
   291	  session.state = "recovering";
   292	  session.ws = null;
   293	
   294	  // Snapshot the generation at the time we start the timer.
   295	  const generationAtStart = session.connectionGeneration;
   296	
   297	  if (session.recoveryTimer !== null) {
   298	    clearTimeout(session.recoveryTimer);
   299	  }
   300

   307	      current.state === "recovering" &&
   308	      current.connectionGeneration === generationAtStart

   315

   450	/**
   451	 * Returns true if the session has exceeded the reconnect rate limit.
   452	 *
   453	 * Limit: 10 reconnects within any rolling 10-minute window.
   454	 * Grace mode: if lastReconnectCause is 'network_drop', allow +5 (limit → 15).
   455	 *
   456	 * reconnectAttempts (lifetime counter) is intentionally NOT used here —

   459	export function isReconnectRateLimited(session: VoiceSession): boolean {
   460	  const windowMs = 10 * 60 * 1000; // 10 minutes
   461	  const cutoff = Date.now() - windowMs;
   462	
   463	  const recentCount = session.reconnectTimestamps.filter(
   464	    (ts) => ts.getTime() > cutoff,
   465	  ).length;
   466	
   467	  const limit = session.lastReconnectCause === "network_drop" ? 15 : 10;
   468	
   469	  return recentCount >= limit;

    58	    .replace(/\*[^*\n]+\*/g, "")
    59	    .replace(/_[^_\n]+_/g, "")
    60	    .replace(/^#{1,6}\s+/gm, "")
    61	    .replace(/^[-–•]\s+/gm, "")
    62	    .replace(/^\d+\.\s+/gm, "")
    63	    .replace(/\*+/g, "")
    64	    .replace(/\n{3,}/g, "\n\n")
    65	    .trim();
    66	}

    68	/** Extract the first complete sentence from buffer. Returns null if none. */
    69	function extractSentence(buf: string): { sentence: string; remainder: string } | null {
    70	  const m = /^([\s\S]+?[.!?])(?:\s+)([\s\S]*)$/.exec(buf);
    71	  if (!m) return null;
    72	  return { sentence: m[1].trim(), remainder: m[2] };
    73	}

    75	async function waitForTtsSlot(): Promise<void> {
    76	  while (ttsInFlight >= CONCURRENT_TTS_POOL_LIMIT) {
    77	    await new Promise<void>((r) => setTimeout(r, TTS_POOL_RETRY_MS));
    78	  }
    79	  ttsInFlight++;
    80	}
    81	
    82	function releaseTtsSlot(): void {
    83	  ttsInFlight = Math.max(0, ttsInFlight - 1);
    84	}

    86	function appendToRollingBuffer(session: VoiceSession, chunk: Buffer): void {
    87	  const combined = Buffer.concat([session.rollingAudioBuffer, chunk]);
    88	  if (combined.byteLength > ROLLING_AUDIO_BUFFER_MAX_BYTES) {
    89	    // Keep the tail (most recent audio).
    90	    session.rollingAudioBuffer = combined.slice(
    91	      combined.byteLength - ROLLING_AUDIO_BUFFER_MAX_BYTES,
    92	    );
    93	  } else {
    94	    session.rollingAudioBuffer = combined;
    95	  }
    96	}

   126	/**
   127	 * Send a pre-generated audio clip if available, otherwise synthesise the
   128	 * fallback text via ElevenLabs. Both paths are best-effort — errors logged,
   129	 * not thrown.
   130	 */
   131	async function speakClipOrText(
   132	  session: VoiceSession,
   133	  clipName: ClipName,
   134	  fallbackText: string,
   135	): Promise<void> {
   136	  if (!session.ws) return;
   137	  if (
   138	    session.state === "closed" ||
   139	    session.state === "failed" ||
   140	    session.state === "closing"
   141	  )
   142	    return;
   143	
   144	  // Save caller's speechId so surrounding pipeline code (e.g. main tts_done)
   145	  // still works correctly after this returns.
   146	  const savedSpeechId = session.currentSpeechId;
   147	  const clipSpeechId  = crypto.randomUUID();
   148	
   149	  // Send speech_start so the client sets ttsCompleteRef=false and clears its
   150	  // chunk buffer. Without this the clip binary gets concatenated with whatever
   151	  // is already buffered (e.g. a previous clip) and plays as one mangled audio.
   152	  try {
   153	    session.ws.send(JSON.stringify({
   154	      type:                "speech_start",
   155	      speechId:             clipSpeechId,
   156	      sessionId:            session.sessionId,
   157	      connectionGeneration: session.connectionGeneration,
   158	      sequenceNumber:       registry.incrementSequence(session),
   159	      timestamp:            Date.now(),
   160	    }));
   161	  } catch {
   162	    return;
   163	  }
   164	
   165	  let sentChunks = 0;
   166	
   167	  if (hasClip(clipName)) {
   168	    try {
   169	      session.ws.send(getClipBuffer(clipName));
   170	      sentChunks = 1;
   171	    } catch (err) {
   172	      logger.warn({ err, clipName }, "VoiceOrch: failed to send clip buffer");
   173	    }
   174	  } else {
   175	    // Text TTS fallback.
   176	    try {
   177	      for await (const chunk of streamSpeechElevenLabs(fallbackText)) {
   178	        if (!session.ws) break;
   179	        try {
   180	          session.ws.send(chunk);
   181	          sentChunks++;
   182	        } catch {
   183	          break;
   184	        }
   185	      }
   186	    } catch {
   187	      // ignore TTS errors for fallback clips
   188	    }
   189	  }
   190	
   191	  // sentence_end flushes the chunk buffer immediately so the clip plays on its
   192	  // own rather than sitting in the buffer until the next sentence_end/tts_done.
   193	  if (sentChunks > 0 && session.ws) {
   194	    try {
   195	      session.ws.send(JSON.stringify({
   196	        type:      "sentence_end",
   197	        speechId:  clipSpeechId,
   198	        timestamp: Date.now(),
   199	      }));
   200	    } catch {}
   201	  }
   202	
   203	  // tts_done signals end of this audio unit. Client sets ttsCompleteRef=true
   204	  // and reopens the mic once playback drains.
   205	  if (session.ws) {
   206	    try {
   207	      session.ws.send(JSON.stringify({
   208	        type:                "tts_done",
   209	        speechId:             clipSpeechId,
   210	        sessionId:            session.sessionId,
   211	        connectionGeneration: session.connectionGeneration,
   212	        sequenceNumber:       registry.incrementSequence(session),
   213	        timestamp:            Date.now(),
   214	      }));
   215	    } catch {}
   216	  }
   217	
   218	  // Restore the caller's speechId ownership.
   219	  session.currentSpeechId = savedSpeechId;
   220	}

   222	/**
   223	 * Stream one TTS sentence to the client and append audio to rolling buffer.
   224	 * Respects speechId ownership — breaks out if cancelled mid-stream.
   225	 */
   226	async function flushSentenceToTTS(
   227	  session: VoiceSession,
   228	  speechId: string,
   229	  sentence: string,
   230	): Promise<void> {
   231	  if (!sentence || session.currentSpeechId !== speechId || !session.ws) return;
   232	
   233	  await waitForTtsSlot();
   234	  let sentChunks = 0;
   235	  try {
   236	    for await (const chunk of streamSpeechElevenLabs(sentence)) {
   237	      if (session.currentSpeechId !== speechId || !session.ws) break;
   238	      appendToRollingBuffer(session, chunk);
   239	      try {
   240	        session.ws.send(chunk);
   241	        sentChunks++;
   242	      } catch {
   243	        break;
   244	      }
   245	    }
   246	  } catch (err: unknown) {
   247	    const isAbort = err instanceof Error && err.name === "AbortError";
   248	    if (!isAbort) {
   249	      logger.warn(
   250	        { err, sessionId: session.sessionId },
   251	        "VoiceOrch: TTS sentence failed — skipping",
   252	      );
   253	    }
   254	  } finally {
   255	    releaseTtsSlot();
   256	  }
   257	
   258	  // Signal sentence boundary so the client can play each sentence as it
   259	  // completes rather than buffering the full turn.
   260	  if (sentChunks > 0 && session.currentSpeechId === speechId && session.ws) {
   261	    try {
   262	      session.ws.send(
   263	        JSON.stringify({ type: "sentence_end", speechId, timestamp: Date.now() }),
   264	      );
   265	    } catch {}
   266	  }
   267	}

   335	  if (session.wasInterrupted || session.rollingAudioBuffer.byteLength === 0) {

   345	    session.ws?.send(session.rollingAudioBuffer);

   347	    logger.warn({ err, sessionId: session.sessionId }, "VoiceOrch: repeat replay failed");
   348	    await speakClipOrText(session, "say-that-again", "Sorry, say that again?");
   349	  }

   352	// ---------------------------------------------------------------------------
   353	// Core LLM + sentence-boundary TTS pipeline
   354	// ---------------------------------------------------------------------------
   355	
   356	async function runLLMAndTTSPipeline(
   357	  session: VoiceSession,
   358	  utterance: string,
   359	  isDirectQuestion: boolean,
   360	): Promise<void> {
   361	  const turnId   = crypto.randomUUID();
   362	  const speechId = crypto.randomUUID();
   363	
   364	  // Capture interruption state before resetting.
   365	  const wasInterrupted    = session.wasInterrupted;
   366	  const remainingResponse = session.remainingResponse;
   367	
   368	  session.currentTurnId       = turnId;
   369	  session.currentResponseId   = crypto.randomUUID();
   370	  session.currentResponseText = "";
   371	  session.llmStreamActive     = true;
   372	  session.state               = "llm_pending";
   373	  session.wasInterrupted      = false;
   374	  session.remainingResponse   = null;
   375	  session.interruptedAt       = null;
   376	
   377	  // P1-1: persist turn start.
   378	  PersistentSessionStateGuard.queue({
   379	    sessionId:            session.sessionId,
   380	    deviceId:             session.deviceId,
   381	    connectionGeneration: session.connectionGeneration,
   382	    state:                session.state,
   383	    currentTurnId:        session.currentTurnId,
   384	    currentResponseId:    session.currentResponseId,
   385	    callStartTime:        session.callStartTime,
   386	    updatedAt:            new Date(),
   387	  });
   388	
   389	  // ── Stage 5: Context Assembly ──────────────────────────────────────────
   390	  let ctx: { systemPrompt: string; messages: { role: "user" | "assistant"; content: string }[] };
   391	  try {
   392	    ctx = await VoiceContextAssembler.assemble({
   393	      sessionId:       session.sessionId,
   394	      deviceId:        session.deviceId,
   395	      callStartTime:   session.callStartTime,
   396	      currentUtterance: utterance,
   397	      reconnectCount:  session.reconnectAttempts,
   398	      isDirectQuestion,
   399	      wasInterrupted,
   400	      remainingResponse,
   401	      passiveMode:     session.passiveMode,
   402	    });
   403	    session.consecutiveContextFailures = 0;
   404	  } catch (err) {
   405	    session.consecutiveContextFailures++;
   406	    logger.error(
   407	      { err, sessionId: session.sessionId, turnId },
   408	      "VoiceOrch: context assembly failed",
   409	    );
   410	    session.llmStreamActive = false;
   411	    session.state           = "listening";
   412	
   413	    if (session.consecutiveContextFailures >= 2) {
   414	      try {
   415	        session.ws?.send(
   416	          JSON.stringify({ type: "call_ended", reason: "context_load_failed" }),
   417	        );
   418	      } catch {}
   419	      registry.finalise(session.sessionId, "context_load_failed");
   420	    } else {
   421	      await speakClipOrText(
   422	        session,
   423	        "say-that-again",
   424	        "Sorry, I lost my thread for a second. Try that again?",
   425	      );
   426	    }
   427	    return;
   428	  }
   429	
   430	  // ── Send speech_start ──────────────────────────────────────────────────
   431	  session.currentSpeechId = speechId;
   432	  session.state           = "tts_streaming";
   433	
   434	  const seqStart = registry.incrementSequence(session);
   435	  try {
   436	    session.ws?.send(
   437	      JSON.stringify({
   438	        type:                "speech_start",
   439	        speechId,
   440	        sessionId:            session.sessionId,
   441	        connectionGeneration: session.connectionGeneration,
   442	        sequenceNumber:       seqStart,
   443	        timestamp:            Date.now(),
   444	      }),
   445	    );
   446	  } catch (err) {
   447	    logger.warn({ err, sessionId: session.sessionId }, "VoiceOrch: failed to send speech_start");
   448	    session.llmStreamActive = false;
   449	    session.state           = "listening";
   450	    session.currentSpeechId = null;
   451	    return;
   452	  }
   453	
   454	  // ── Stage 6: LLM streaming with watchdogs ──────────────────────────────
   455	  const controller = new AbortController();
   456	  session.currentAbortController = controller;
   457	
   458	  let firstTokenReceived         = false;
   459	  let firstTokenWatchdog: NodeJS.Timeout | null = null;
   460	
   461	  const hardBudget = setTimeout(() => {
   462	    if (session.currentAbortController === controller) {
   463	      logger.warn({ sessionId: session.sessionId, turnId }, "VoiceOrch: hard budget exceeded — aborting");
   464	      registry.cancelCurrentTurn(session, "hard_budget_exceeded");
   465	      session.llmStreamActive = false;
   466	
   467	      // Notify the client immediately so it re-enters listening state rather
   468	      // than hanging indefinitely while the LLM stream drains its abort.
   469	      if (session.currentSpeechId === speechId) {
   470	        const seqBudget = registry.incrementSequence(session);
   471	        try {
   472	          session.ws?.send(
   473	            JSON.stringify({
   474	              type:                "tts_done",
   475	              speechId,
   476	              responseText:         session.currentResponseText ?? "",
   477	              sessionId:            session.sessionId,
   478	              connectionGeneration: session.connectionGeneration,
   479	              sequenceNumber:       seqBudget,
   480	              timestamp:            Date.now(),
   481	              reason:               "budget_exceeded",
   482	            }),
   483	          );
   484	        } catch {}
   485	        session.currentSpeechId = null;
   486	      }
   487	    }
   488	  }, MAX_TURN_BUDGET_MS);
   489	
   490	  firstTokenWatchdog = setTimeout(() => {
   491	    if (!firstTokenReceived && session.currentTurnId === turnId) {
   492	      logger.info({ sessionId: session.sessionId, turnId }, "VoiceOrch: first-token watchdog — playing thinking clip");
   493	      void speakClipOrText(session, "thinking", "Hang on, I'm thinking...");
   494	    }
   495	  }, FIRST_TOKEN_WATCHDOG_MS);
   496	
   497	  let sentenceBuffer = "";
   498	
   499	  try {
   500	    const stream = streamChatText({
   501	      system:        ctx.systemPrompt,
   502	      messages:      ctx.messages,
   503	      maxTokens:     VOICE_MAX_TOKENS,
   504	      forceProvider: "gemini",
   505	      signal:        controller.signal,
   506	    });
   507	
   508	    for await (const chunk of stream) {
   509	      // Bail if the turn was cancelled (interrupt or abort).
   510	      if (session.currentTurnId !== turnId || !session.llmStreamActive) break;
   511	
   512	      if (!firstTokenReceived) {
   513	        firstTokenReceived = true;
   514	        if (firstTokenWatchdog) {
   515	          clearTimeout(firstTokenWatchdog);
   516	          firstTokenWatchdog = null;
   517	        }
   518	      }
   519	
   520	      session.currentResponseText += chunk;
   521	      sentenceBuffer              += chunk;
   522	
   523	      // Stage 7: flush complete sentences to TTS as they arrive.
   524	      let extracted = extractSentence(sentenceBuffer);
   525	      while (extracted) {
   526	        if (session.currentSpeechId !== speechId) break;
   527	        const { sentence, remainder } = extracted;
   528	        sentenceBuffer = remainder;
   529	        const clean = stripMarkdown(sentence);
   530	        if (clean) {
   531	          await flushSentenceToTTS(session, speechId, clean);
   532	        }
   533	        extracted = extractSentence(sentenceBuffer);
   534	      }
   535	    }
   536	
   537	    // Flush any trailing partial sentence.
   538	    const trail = sentenceBuffer.trim();
   539	    if (trail && session.currentSpeechId === speechId) {
   540	      await flushSentenceToTTS(session, speechId, stripMarkdown(trail));
   541	    }
   542	  } catch (err: unknown) {
   543	    // Treat both the Web standard AbortError and the Anthropic SDK's
   544	    // APIUserAbortError as expected cancellations (budget exceeded, interrupt).
   545	    const isAbort =
   546	      err instanceof Error &&
   547	      (err.name === "AbortError" || err.name === "APIUserAbortError");
   548	    if (!isAbort) {
   549	      logger.error({ err, sessionId: session.sessionId, turnId }, "VoiceOrch: LLM stream error");
   550	      await speakClipOrText(
   551	        session,
   552	        "say-that-again",
   553	        "Sorry, I've lost my train of thought. What were you saying?",
   554	      );
   555	    }
   556	  } finally {
   557	    clearTimeout(hardBudget);
   558	    if (firstTokenWatchdog) clearTimeout(firstTokenWatchdog);
   559	    if (session.currentAbortController === controller) {
   560	      session.currentAbortController = null;
   561	    }
   562	    session.llmStreamActive = false;
   563	  }
   564	
   565	  // ── Send tts_done if we still own the speechId ─────────────────────────
   566	  if (session.currentSpeechId === speechId) {
   567	    const seqDone = registry.incrementSequence(session);
   568	    try {
   569	      session.ws?.send(
   570	        JSON.stringify({
   571	          type:                "tts_done",
   572	          speechId,
   573	          responseText:         session.currentResponseText,
   574	          sessionId:            session.sessionId,
   575	          connectionGeneration: session.connectionGeneration,
   576	          sequenceNumber:       seqDone,
   577	          timestamp:            Date.now(),
   578	        }),
   579	      );
   580	    } catch {}
   581	    session.currentSpeechId = null;
   582	  }
   583	
   584	  // Store last response for REPEAT_REQUEST.
   585	  session.lastResponseBuffer = session.currentResponseText;
   586	
   587	  // Persist assistant row (fire-and-forget).
   588	  if (session.currentResponseText.trim() && !session.completedTurnIds.has(turnId)) {
   589	    session.completedTurnIds.add(turnId);
   590	    persistMessage(session.deviceId, "ashley", session.currentResponseText, new Date());
   591	  }
   592	
   593	  session.state         = "listening";
   594	  session.currentTurnId = null;
   595	  session.currentResponseId = null;
   596	}

   602	/**
   603	 * Start zombie cleanup for a session. Called once per connection (new or
   604	 * reclaimed). The interval self-clears when the session leaves the registry.
   605	 */
   606	export function startOrchestration(session: VoiceSession, ws: WsLike): void {
   607	  // Clear any stale handle for this session (e.g., rapid reconnect).
   608	  const existing = zombieHandles.get(session.sessionId);
   609	  if (existing) clearInterval(existing);
   610	
   611	  const handle = setInterval(() => {
   612	    const live = registry.findBySessionId(session.sessionId);
   613	    if (!live) {
   614	      clearInterval(handle);
   615	      zombieHandles.delete(session.sessionId);
   616	      return;
   617	    }
   618	
   619	    const elapsed = Date.now() - live.lastActiveHeartbeatAt.getTime();
   620	    if (elapsed > ZOMBIE_TIMEOUT_MS) {
   621	      logger.warn(
   622	        { sessionId: live.sessionId, elapsedMs: elapsed },
   623	        "VoiceOrch: zombie session detected — closing",
   624	      );
   625	      try {
   626	        ws.close();
   627	      } catch {}
   628	      registry.finalise(live.sessionId, "zombie_cleanup");
   629	      clearInterval(handle);
   630	      zombieHandles.delete(live.sessionId);
   631	    }
   632	  }, ZOMBIE_CLEANUP_INTERVAL_MS);
   633	
   634	  zombieHandles.set(session.sessionId, handle);
   635	}

   637	/**
   638	 * Stage 2: Handle a speech_interim message.
   639	 * Runs intent pre-classification; sends set_silence_threshold to client if
   640	 * a direct question is detected. Does NOT trigger LLM or TTS.
   641	 */
   642	export function handleSpeechInterim(
   643	  session: VoiceSession,
   644	  transcript: string,
   645	  ws: WsLike,
   646	): void {
   647	  // Update heartbeat.
   648	  session.lastActiveHeartbeatAt = new Date();
   649	
   650	  // Stage 8A: interruption check.
   651	  if (session.llmStreamActive || session.state === "tts_streaming") {
   652	    handleInterruption(session);
   653	  }
   654	
   655	  const result = classify(transcript.trim());
   656	
   657	  if (result.category === "DIRECT_QUESTION" && !session.intentDetected) {
   658	    session.intentDetected   = true;
   659	    session.silenceThreshold = 800;
   660	    try {
   661	      ws.send(JSON.stringify({ type: "set_silence_threshold", ms: 800 }));
   662	    } catch {}
   663	  }
   664	}

   666	/**
   667	 * Stage 3-7: Handle a speech_final message — the full turn pipeline.
   668	 * Called from handleVoiceTurn in routes/voice-call.ts.
   669	 */
   670	export async function handleSpeechFinal(
   671	  session: VoiceSession,
   672	  transcript: string,
   673	  utteranceId: string,
   674	): Promise<void> {
   675	  // Update heartbeat.
   676	  session.lastActiveHeartbeatAt = new Date();
   677	
   678	  // Stage 8A: interruption check.
   679	  if (session.llmStreamActive || session.state === "tts_streaming") {
   680	    handleInterruption(session);

   299	      // Update silence tracking when real audio arrives (1I).
   300	      currentSession.lastAudioReceivedAt = new Date();
   301	      currentSession.silenceWarningSent = false;
   302	      // Audio forwarding to Deepgram wired in 1G.
   303	      return;
   304	    }
   305	
   306	    // Text frame — JSON control message.
   307	    let msg: Record<string, unknown>;
   308	    try {
   309	      msg = JSON.parse(
   310	        Buffer.isBuffer(data) ? data.toString("utf8") : String(data),
   311	      ) as Record<string, unknown>;
   312	    } catch {
   313	      logger.warn({ deviceId }, "Voice-call WS: non-JSON text message ignored");
   314	      return;
   315	    }
   316	
   317	    const msgSessionId = msg["sessionId"] as string | undefined;
   318	    const msgGen = msg["connectionGeneration"] as number | undefined;
   319	
   320	    if (
   321	      msgSessionId !== undefined &&
   322	      msgGen !== undefined &&
   323	      !registry.validateMessage(currentSession, {
   324	        sessionId: msgSessionId,
   325	        connectionGeneration: msgGen,
   326	      })
   327	    ) {
   328	      return; // stale message — already logged by validateMessage
   329	    }
   330	
   331	    // P1-4 Stage 2: speech_interim — intent pre-classification only.
   332	    if (msg["type"] === "speech_interim") {
   333	      const interimTranscript = ((msg["transcript"] as string | undefined) ?? "").trim();
   334	      if (interimTranscript) {
   335	        VoiceOrchestrationService.handleSpeechInterim(currentSession, interimTranscript, ws as registry.WsLike);
   336	      }
   337	      return;
   338	    }
   339	
   340	    // P1-4: call_end — client-initiated graceful call termination.
   341	    if (msg["type"] === "call_end") {
   342	      logger.info({ deviceId, sessionId: currentSession.sessionId }, "voice: client sent call_end");
   343	      try {
   344	        ws.send(JSON.stringify({ type: "call_ended", reason: "client_call_end" }));
   345	      } catch {}
   346	      registry.finalise(currentSession.sessionId, "client_call_end");
   347	      return;
   348	    }
   349	
   350	    if (msg["type"] === "speech_final") {
   351	      const transcript = (msg["transcript"] as string | undefined)?.trim() ?? "";
   352	      const utteranceId = (msg["utteranceId"] as string | undefined) ?? crypto.randomUUID();
   353	
   354	      if (!transcript) return;
   355	
   356	      // RACE 4: Deepgram finals must not be processed while the session is closing.
   357	      if (
   358	        currentSession.state === "closing" ||
   359	        currentSession.state === "closed" ||
   360	        currentSession.state === "failed"
   361	      ) {
   362	        logger.info(
   363	          { deviceId, state: currentSession.state },
   364	          "voice: speech_final ignored — session is closing",
   365	        );
   366	        return;
   367	      }
   368	
   369	      // Idempotency: ignore duplicate utteranceIds.
   370	      if (currentSession.processedUtteranceIds.has(utteranceId)) {
   371	        logger.info({ deviceId, utteranceId }, "voice: duplicate utteranceId ignored");
   372	        return;
   373	      }
   374	      currentSession.processedUtteranceIds.add(utteranceId);
   375	
   376	      logger.info({ deviceId, utteranceId, transcriptPreview: transcript.slice(0, 80) }, "voice: speech_final accepted");
   377	
   378	      // handleVoiceTurn runs the full pipeline: context → Claude → TTS.
   379	      // TTS, DB writes, and state reset happen inside the function.
   380	      await handleVoiceTurn(currentSession, transcript, utteranceId);
   381	    }
   382	
   383	    if (msg["type"] === "interrupt") {
   384	      // 1G: User interrupts Ashley mid-speech.
   385	      // cancelCurrentTurn is called BEFORE sending interrupt_ack so that
   386	      // any in-flight TTS chunk send sees the cleared speechId first.
   387	      registry.cancelCurrentTurn(currentSession, "user_interrupt");
   388	      currentSession.state = "listening";
   389	
   390	      const seq = registry.incrementSequence(currentSession);
   391	      try {
   392	        ws.send(
   393	          JSON.stringify({
   394	            type: "interrupt_ack",
   395	            sessionId: currentSession.sessionId,
   396	            connectionGeneration: currentSession.connectionGeneration,
   397	            sequenceNumber: seq,
   398	            timestamp: Date.now(),
   399	          }),
   400	        );
   401	      } catch (err) {
   402	        logger.warn({ err, deviceId }, "voice: failed to send interrupt_ack");
   403	      }
   404	      logger.info(
   405	        { deviceId, sessionId: currentSession.sessionId },
   406	        "voice: interrupt handled",
   407	      );
   408	    }
   409	  });
   410	
   411	  ws.on("close", (code: number, reason: Buffer) => {
   412	    logger.info(
   413	      { deviceId, sessionId, code, reason: reason.toString() },
   414	      "Voice-call WS: closed — marking recovering",
   415	    );
   416	    const closingSession = registry.findBySessionId(sessionId);
   417	    if (closingSession) {
   418	      // P1-3: detect cause from WS close code before marking recovering.
   419	      if (code === 1000 || code === 1001) {
   420	        closingSession.lastReconnectCause = "clean_close";
   421	      } else if (code === 1006) {
   422	        // 1006 = abnormal closure, no close frame — network drop.
   423	        closingSession.lastReconnectCause = "network_drop";
   424	      } else {
   425	        closingSession.lastReconnectCause = "timeout";
   426	      }
   427	      registry.markRecovering(sessionId);
   428	    }
   429	  });
   430	
   431	  ws.on("error", (err: Error) => {
   432	    logger.error({ err, deviceId, sessionId }, "Voice-call WS: error");
   433	  });
   434	});
